#include <iostream>
#include <assert.h>
using namespace std;
void reverse(string &str,bool flag,int m,int beg,int end,int legth)
{
	assert(&str!=nullptr);
	if (beg==end||m<0)
	{
		return;
	}
	if (flag==true)
	{
		int p1=beg;
		int p2=beg+m;
		int k=legth-m-legth%m;
		while (k)
		{
			swap(str[p1],str[p2]);
			p1++;
			p2++;
			k--;
		}
	    k=legth-m-legth%m;
		reverse(str,false,legth%m,p1,end,legth-k);
	}
	else
	{
		int p1=end-m;
		int p2=end;
		int k=legth-m-legth%m;
		while(k)
		{
			swap(str[p1],str[p2]);
			p2--;
			p1--;
			k--;
		}
		k=legth-m-legth%m;
		reverse(str,true,legth%m,p1,p2,legth-k);
	}
}
int main()
{
	string str="abcdefghijk";
	reverse(str,true,3,0,10,11);
	cout<<str.c_str()<<endl;
	return 0;
}
